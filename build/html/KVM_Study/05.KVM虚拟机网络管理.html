5.2. 桥接网络配置¶

1.设置桥接网络

[root@kvm ~]# virsh iface-bridge eth0 br0
使用附加设备 br0 生成桥接 eth0 失败
已启动桥接接口 br0

2.查看网卡配置文件

#查看eth0 配置文件
[root@bogon network-scripts]# cat ifcfg-eth0
DEVICE="eth0"
NM_CONTROLLED=NM_CONTROLLED="no"
ONBOOT="yes"
BRIDGE="br0"

#查看br0 配置文件
[root@bogon network-scripts]# cat ifcfg-br0
DEVICE="br0"
ONBOOT="yes"
TYPE="Bridge"
BOOTPROTO="dhcp"
STP="on"
DELAY="0"
  1. 修改虚拟机网络配置

virsh edit centos7

#修改前：
   <interface type='network'>
      <mac address='52:54:00:42:bf:bc'/>
      <source network='default'/>
      <model type='virtio'/>
      <address type='pci' domain='0x0000' bus='0x00' slot='0x03' function='0x0'/>
    </interface>

#修改后
    <interface type='bridge'>
      <mac address='52:54:00:41:46:eb'/>
      <source bridge='br0'/>
      <model type='virtio'/>
      <address type='pci' domain='0x0000' bus='0x00' slot='0x03' function='0x0'/>
    </interface>
